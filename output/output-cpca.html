# Commuting Matrices: http://mathworld.wolfram.com/CommutingMatrices.html

library(JADE)
## Loading required package: clue
library(abind)

library(plyr)

### run example in help section `?djd`
set.seed(1) # for reproducibility of results

Z <- matrix(runif(9), ncol = 3)
U <- eigen(Z %*% t(Z))$vectors

D1 <- diag(runif(3))
D2 <- diag(runif(3))
D3 <- diag(runif(3))
D4 <- diag(runif(3))

X1 <- t(U) %*% D1 %*% U
X2 <- t(U) %*% D2 %*% U
X3 <- t(U) %*% D3 %*% U
X4 <- t(U) %*% D4 %*% U

# check matrices `X1` and `X2` commute
X1 %*% X2
##          [,1]     [,2]     [,3]
## [1,]  0.08436  0.01553 -0.04050
## [2,]  0.01553  0.09137 -0.01765
## [3,] -0.04050 -0.01765  0.08175
X2 %*% X1
##          [,1]     [,2]     [,3]
## [1,]  0.08436  0.01553 -0.04050
## [2,]  0.01553  0.09137 -0.01765
## [3,] -0.04050 -0.01765  0.08175

# `X1` and `X2` commute, as their product is like `D2 %*% D1` (D1, D2 are diagonal)
D1 %*% D2
##         [,1]    [,2]   [,3]
## [1,] 0.04245 0.00000 0.0000
## [2,] 0.00000 0.07912 0.0000
## [3,] 0.00000 0.00000 0.1359
D2 %*% D1
##         [,1]    [,2]   [,3]
## [1,] 0.04245 0.00000 0.0000
## [2,] 0.00000 0.07912 0.0000
## [3,] 0.00000 0.00000 0.1359

# let's start Joint Diagonalization
X <- abind(X1, X2, X3, X4, along = 3) # a nice way to combine several 2D matricies into one 3D matrix
dim(X) # should be a 3D array
## [1] 3 3 4

W <- djd(X) 
# W <- djd(X, r = 1)
# W <- djd(X, G = "l")

# Common PC in weight matrix `W` are exactly vectors in transformation matirx `U` (used to generate X1, X2, ...)
round(U %*% W, 4) # should be a signed permutation matrix if W1 is correct.
##      [,1] [,2] [,3]
## [1,]    0    0   -1
## [2,]    1    0    0
## [3,]    0    1    0

library(JADE)

### data
data(iris)

X <- iris[, 1:2]

### prepare corr. matrices

# check scaling does not affect corr.
cor(X)
##              Sepal.Length Sepal.Width
## Sepal.Length       1.0000     -0.1176

cor(scale(X))

# scale data

# compute corr. matricies `C`
names(cbind(X, Y))
## NULL

C <- daply(data.frame(X, Y), "Y", function(x) cor(x[, -ncol(x)]))
